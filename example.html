Usage Example

// Import the library
import { VideoConverter, FileHandler } from './dist/main.js';

// Create converter
const converter = new VideoConverter();

// Set event listeners
converter.setListeners({
  onProgress: (progress) => {
    console.log(`${progress.percentage}%: ${progress.message}`);
  },
  onComplete: (result) => {
    // Download files
    FileHandler.downloadAudio(...);
    FileHandler.downloadTranscript(...);
  }
});

// Convert video
const config = VideoConverter.createDefaultConfig('webm');
await converter.convert(videoFile, config);
